テスト


using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;
using UniRx;

public class EXPManager : MonoBehaviour
{
    [SerializeField]
    private CameraSize cameraSize;

    [SerializeField]
    private Image fill_EXP_Image;

    [SerializeField]
    private Image gage_EXP_Image;

    [SerializeField,Tooltip("High_EXP入れてね")]
    private Transform lastImage;

    private RectTransform rect;

    [SerializeField]
    private BoxCollider2D boxCollider;

    [SerializeField]
    private List gageList = new List();

    [SerializeField]
    private List fillList = new List();

    private const int IMAGE_WIDTH = 50;
    private const int IMAGE_HEIGHT = 10;

    private void Awake()
    {
        rect = GetComponent();
        DontDestroyOnLoad(transform.parent.gameObject);
    }

    private void Start()
    {
        //登録処理
        PlayerController.Instance.Level.Subscribe(x =>OnLevelChenge(x));
    }

    public void OnLevelChenge(int level)
    {
        while (level >= gageList.Count)
        {
            if (level > gageList.Count)
            {
                gageList.Add(InstantiateGage());

                if((gageList.Count -1) == fillList.Count)
                {

                }
                else
                {
                    fillList.Add(InstantiateFill());
                }
            }
            else
            {
                break;
            }
        }
        //縦幅計算式 (ゲージの数+最初から存在してる縁Image(2個)) * Image縦幅
        rect.sizeDelta = new Vector2(IMAGE_WIDTH, (gageList.Count + fillList.Count + 2) * IMAGE_HEIGHT);
        boxCollider.size = rect.sizeDelta;
        boxCollider.offset = new Vector2(0, boxCollider.size.y / 2);
        //縁を一番最後に移動
        lastImage.SetAsLastSibling();

        if(level == 10)
        {
            cameraSize.LevelTen();
            transform.localScale /= 2;
        }
        else if(level == 20)
        {
            cameraSize.LevelTwenty();
            transform.localScale /= 1.5f;
        }
    }
    private Image InstantiateGage()
    {
        var obj = Instantiate(gage_EXP_Image, transform.position, new Quaternion(0,0,0,0), transform);
        return obj;
    }

    private Image InstantiateFill()
    {
        var obj = Instantiate(fill_EXP_Image, transform.position, new Quaternion(0, 0, 0, 0), transform);
        return obj;
    }
}
